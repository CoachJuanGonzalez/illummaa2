  max: 2, // 2 submissions per IP
  skipSuccessfulRequests: true,
  standardHeaders: true,

  message: { error: 'Security limit: Too many attempts. Please wait 10 minutes.' }

// Main assessment handler
export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ 
      success: false,
      error: 'Method not allowed' 
    });
  }

  try {
    // Apply SMS-specific rate limiting
    await new Promise((resolve, reject) => {
      smsConsentLimiter(req, res, (result) => {
        if (result instanceof Error) reject(result);
        else resolve(result);
      });
    });

    // Apply general rate limiting
    await new Promise((resolve, reject) => {
      strictLimiter(req, res, (result) => {
        if (result instanceof Error) reject(result);
        else resolve(result);
      });
    });

    // CSRF validation
    await new Promise((resolve, reject) => {
      csrfProtection(req, res, (result) => {
        if (result instanceof Error) reject(result);
        else resolve(result);
      });
    });

    // Enhanced validation including SMS security
    await Promise.all(secureValidation.map(validation => validation.run(req)));
    const errors = validationResult(req);
    
    if (!errors.isEmpty()) {
      console.warn('SMS consent security validation failed:', {
        ip: req.ip,
        errors: errors.array(),
